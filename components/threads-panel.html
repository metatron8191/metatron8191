// In app.js, add or verify this function exists window.createNewThread = async function() { if (!window.fabric) { console.error('Fabric not initialized'); alert('Fabric not ready'); return; } const threadName = prompt('New thread name:', `Thread ${new Date().toLocaleString()}`); if (!threadName) return; try { // Save current thread messages const currentThreadId = window.fabric.currentThreadId; if (currentThreadId && window.memory) { const allMessages = [...(window.memory.active || []), ...(window.memory.working || [])]; for (const msg of allMessages) { await window.fabric.addMemory(msg.content, msg.role, msg.model || 'unknown', { threadId: currentThreadId, timestamp: msg.timestamp || Date.now() }); } } // Create new thread const newThread = await window.fabric.createThread(threadName); await window.fabric.switchThread(newThread.id); // Clear conversation history for new thread window.conversationHistory[newThread.id] = []; // Clear UI if (window.memory) { window.memory.clearWorking(); window.memory.active = []; } if (window.UIRenderer) window.UIRenderer.renderMessages(); if (window.updateRightPanelMemoryLists) window.updateRightPanelMemoryLists(); // Refresh thread lists if (window.refreshSidebarThreadList) window.refreshSidebarThreadList(); if (window.refreshFabricThreadList) window.refreshFabricThreadList(); if (window.updateChainDisplay) window.updateChainDisplay(); console.log(`✨ Created new thread: ${threadName}`); } catch (e) { console.error('Failed to create thread:', e); alert('Failed to create thread: ' + e.message); } };
